    """ A class that handles the logic for creating transition and final state guesses from initial structure. 

    Note:
        1. ads1 and ads2 corresponds to the adsorbate derived from the atom at diss_idx[0] and diss_idx[1] respectively.
        2. By default, ads1_dist_filter and ads1_dir_filter is True while ads2_dist_filiter and ads2_dir filter is False. 
            (i.e., ads1 is programmed to be moved away from the adsorbate to the specified site while ads2 is programmed to find the nearest site.)
        3. If ads1 binding site == ads2 binding site, ads1 would be given priority for the site. 
        4. If ads1/ads2 is bidentate, the exisiting bond on the slab would be prioritized.

    Attributes:
        initial_file (str): File path to initial structure.
        diss_idx (list): List of len 2 that specifies the index of atoms to be dissociated (i.e. [ads1_idx, ads2_idx])
        slab_elements (list): List indicating the element composition of the slab (i.e., ['Rh','Pd']). If 'None' specified LaspBuilder will assume transition metals as the slab elements. Defaults to None.

        ads1_acn (list, int, float): Specifies avg_coord_num of site for ads1. Defaults to None.
        ads1_connectivity (list, int): Specifies connectivity of site for ads1. Defaults to None.
        ads1_bind (bool): Toggle to bind ads1 to site. The position of ads1 will remain unchanged if False. Defaults to True.
        ads1_dist_tol (float): XY distance tolerance (in angstroms) for ads1 adsorption site selection. Defaults to 1.5.
        ads1_dir_tol (float): XY vector direction tolerance for ads1 adsorption site selection. Where -1 <= directon_tol <= 1. Defaults to 0.5.
        ads1_dist_filter (bool): Toggle to filter off sites that are too close to ads1 initial position. Defaults to True.
        ads1_dir_filter (bool): Toggle to filter off sites that are not facing the same direction as the dissociation vector. Defaults to True.

        ads2_acn (list, int, float): Specifies avg_coord_num of site for ads2. Defaults to None.
        ads2_connectivity (list, int): Specifies connectivity of site for ads2. Defaults to None.
        ads2_bind (bool): Toggle to bind ads2 to site. The position of ads2 will remain unchanged if False. Defaults to True.
        ads2_dist_tol (float): XY distance tolerance (in angstroms) for ads2 adsorption site selection. Defaults to 1.5.
        ads2_dir_tol (float): XY vector direction tolerance for ads2 adsorption site selection. Where -1 <= directon_tol <= 1. Defaults to 0.5.
        ads2_dist_filter (bool): Toggle to filter off sites that are too close to ads2 initial position. Defaults to False.
        ads2_dir_filter (bool): Toggle to filter off sites that are not facing the same direction as the dissociation vector. Defaults to False.


        zoom (bool): Toggle to return initial and final states that are closer to the transition state. Defaults to True.
        zoom_tol (float): bond distance between initial, transition and final state. (i.e., if bond distance(TS)=1.5 and zoom_tol=0.3, bond distance(IS)=1.2 and bond_distance(FS)=1.8). Defaults to 0.3.
    """

    @property
    def slab_idx(self) -> List[int]:
        """ index of slab """
        if self.slab_elements:
            slab_atomic_numbers = Atoms(self.slab_elements).get_atomic_numbers()
        else:
            # selects transition metals as slab atoms by default
            slab_atomic_numbers = chain(
                range(21, 31), range(39, 49), range(72, 81), range(104, 109))

        slab_idx = [list(self.initial.find_atomic_number(atomic_number=i)) for i in slab_atomic_numbers]
        slab_idx = list(chain.from_iterable(slab_idx))
        return slab_idx

    @property
    def ads_idx(self) -> List[int]:
        """ List of atom indexes of the adsorbate.

        Raises:
            ValueError: If diss_idx[i] not in ads_idx.
        """
        indices = list(range(len(self.initial)))
        ads_idx = list(set(indices) - set(self.slab_idx))

        if any([i not in ads_idx for i in self.diss_idx]):
            raise ValueError("Invalid diss_idx! diss_idx must be adsorbate atom index")
        return ads_idx

    @property

    @property
    def filtered_sites(self):
        # sites for ads1 and ads2
        filtered_sites = [self.sites, self.sites]
        filtered_sites = [sites.sort_sites_by_distances(pos)
                          for sites, pos in zip(filtered_sites, self.atoms_init_pos)]

        filtered_sites = [sites if c is None else sites.filter_connectivities(c)
                          for sites, c in zip(filtered_sites, self.connectivity)]
        filtered_sites = [sites if acn is None else sites.filter_avg_coord_num(acn)
                          for sites, acn in zip(filtered_sites, self.avg_coord_num)]
        filtered_sites = [sites if filter is None else sites.filter_direction(point=pos, vec=vec, dir_tol=self.direction_tol)
                          for sites, filter, pos, vec in zip(filtered_sites, self.direction_filter, self.atoms_init_pos, self.ads_diss_vec)]
        filtered_sites = [sites if filter is None else sites.filter_distances(point=pos, cutoff=self.distance_tol)
                          for sites, filter, pos in zip(filtered_sites, self.distance_filter, self.atoms_init_pos)]

        return filtered_sites

    @property
    def atoms_init_pos(self) -> List[np.ndarray]:
        """ Initial position of atoms before bond breaking. """
        return [self.initial[self.diss_idx[0]].position, self.initial[self.diss_idx[1]].position]

    @property
    def ads_diss_vec(self) -> List[np.ndarray]:
        """ Direction vector of the dissociation of ads1 and ads2."""
        ads_diss_vec = self.initial.get_distance(
            self.diss_idx[0], self.diss_idx[1], vector=True, mic=True)
        return [ads_diss_vec, -ads_diss_vec]

    @property
    def adsorption_schematic(self) -> AtomsWrapper:
        """Returns decorated slab showing adsorption schematic of ads1 and ads2 based on initialization criteria.
        Site(s) for ads1 ands ads2 represented by 'X' and 'Ne' respectively.

        Raises:
            ValueError: If adsorbate splitting failed
        """
        # get ads1 and ads2 from ads using graph theory
        mapped_dis_idx = [self.ads_idx.index(self.diss_idx[0]), self.ads_idx.index(self.diss_idx[1])]

        graph = self.adsorbate.graph(neighbour_distance=1.6)
        broken_graph = graph.delete_edges([tuple(mapped_dis_idx)])
        graph_components = broken_graph.components()

        # check if ads is split to 2 components
        if len(graph_components) != 2:
            raise ValueError("Adsorbate splitting failed")

        if mapped_dis_idx[0] in graph_components[1] and mapped_dis_idx[1] in graph_components[0]:
            ads1_idx, ads2_idx = graph_components[1], graph_components[0]
            ads1_new_bond = graph_components[1].index(mapped_dis_idx[0])
            ads2_new_bond = graph_components[0].index(mapped_dis_idx[1])

        elif mapped_dis_idx[0] in graph_components[0] and mapped_dis_idx[1] in graph_components[1]:
            ads1_idx, ads2_idx = graph_components[0], graph_components[1]
            ads1_new_bond = graph_components[0].index(mapped_dis_idx[0])
            ads2_new_bond = graph_components[1].index(mapped_dis_idx[1])
        else:
            raise ValueError("Adsorbate splitting failed")

        ads1 = self.adsorbate[ads1_idx].copy()
        ads2 = self.adsorbate[ads2_idx].copy()

        # get denticity of ads1 and ads2 using graph theory
        from catplat.structure.adsorbate import Adsorbate
        ads1_bond = Adsorbate(atoms=ads1, connectivity=self.ads1_connectivity, avg_coord_num=self.ads1_acn).get_bonds()
        ads2_bond = Adsorbate(atoms=ads2, connectivity=self.ads2_connectivity, avg_coord_num=self.ads2_acn).get_bonds()

        if len(ads1_bond) == 0:
            ads1_bond = Adsorbate(atoms=ads1, connectivity=self.ads1_connectivity,
                                  avg_coord_num=self.ads1_acn).get_bonds(length=1)
        if len(ads2_bond) == 0:
            ads2_bond = Adsorbate(atoms=ads2, connectivity=self.ads2_connectivity,
                                  avg_coord_num=self.ads2_acn).get_bonds(length=1)

        # get site idx of ads1 and ads2
        ads1_site, ads2_site = self._get_ads_site(ads1=ads1, ads1_bond=ads1_bond, ads1_new_bond=ads1_new_bond,
                                                  ads2=ads2, ads2_bond=ads2_bond, ads2_new_bond=ads2_new_bond)

        coordinates = self.sites.coordinates
        schematic = self.initial.copy()

        if isinstance(ads1_site, list):  # list of len 2 for bidentate
            for i in ads1_site:
                schematic += Atom('X', coordinates[i] + [0, 0, 5])
        else:  # integer for monodentate
            schematic += Atom('X', coordinates[ads1_site] + [0, 0, 5])

        if isinstance(ads2_site, list):
            for i in ads2_site:
                schematic += Atom('Ne', coordinates[i] + [0, 0, 5])
        else:
            schematic += Atom('Ne', coordinates[ads2_site] + [0, 0, 5])
        return schematic

    def run(self) -> AtomsWrapper:
        """ Main function to get initial, transition and final states. 

        Returns:
            3 AtomsWrapper objects: initial, transition and final state structures

        """
        initial_state = self.initial.copy()
        final_state = self._get_fs()

        if self.zoom:
            initial_state, transition_state, final_state = self._get_ts(initial_state, final_state)
            return initial_state, transition_state, final_state

        transition_state = self._get_ts(
            initial_state=initial_state, final_state=final_state)

        return initial_state, transition_state, final_state

    def _filter_sites(self, point, args_preset, n=0) -> int:
        """Filter adsorption sites and return site idx of n'th closest site that matches filtered criteria

        Args:
            point (list): Coordinates of reference point.
            args_preset (str): Preset for function args. Args can be 'ads1 monodentate', 'ads2 monodentate', 'ads1 bidentate', 'ads2 bidentate', 'closest site'.
            n (int): n'th closest to return after filtering sites.

        Raises:
            ValueError if invalid args_preset is defined.
            ValueError if no sites are found after filtering i.e., len(indices) = 0 

        Returns:
            List index of selected site.
        """

        # preset args for ads1 monodentate

        direction_filter, distance_filter = False, False
        direction_point, direction_vec, direction_tol, distance_cutoff = None, None, None, None

        if args_preset == 'ads1 monodentate':
            connectivity = self.ads1_connectivity
            avg_coord_num = self.ads1_acn
            direction_filter = self.ads1_dir_filter
            direction_point = self.ads2_init_pos
            direction_vec = self.ads1_diss_vec
            direction_tol = self.ads1_dir_tol
            distance_filter = self.ads1_dist_filter
            distance_cutoff = self.ads1_dist_tol

        # preset args for ads2 monodentate
        elif args_preset == 'ads2 monodentate':
            connectivity = self.ads2_connectivity
            avg_coord_num = self.ads2_acn

            direction_filter = self.ads2_dir_filter
            direction_point = self.ads1_init_pos
            direction_vec = self.ads2_diss_vec
            direction_tol = self.ads2_dir_tol

            distance_filter = self.ads2_dist_filter
            distance_cutoff = self.ads2_dist_tol

        # preset args for new bond for ads1 bidentate
        elif args_preset == 'ads1 bidentate':
            connectivity = self.ads1_connectivity
            avg_coord_num = self.ads1_acn

        # preset args for new bond for ads2 bidentate
        elif args_preset == 'ads2 bidentate':
            connectivity = self.ads2_connectivity
            avg_coord_num = self.ads2_acn

        # preset args for existing bond for ads1/ads2 bidentate
        elif args_preset == 'closest site':
            connectivity = None
            avg_coord_num = None

        else:
            raise ValueError('Invalid preset for _filter_sites')

        sites = self.sites
        sites = sites.filter(sort_point=point,
                             unique=False,
                             connectivity=connectivity,
                             avg_coord_num=avg_coord_num,
                             distance_filter=distance_filter,
                             distance_point=point,
                             distance_cutoff=distance_cutoff,
                             direction_filter=direction_filter,
                             direction_point=direction_point,
                             direction_vec=direction_vec,
                             direction_tol=direction_tol
                             )
        indices = sites.index
        if len(indices) == 0:
            raise ValueError(
                "No site suitable site found, consider broadening filtered criteria")
        return indices[n]  # n'th closest site

    def _get_fs(self) -> AtomsWrapper:
        """Derives the final state guess structure. 

        Returns:
            AtomsWrapper: Final state guess structure

        Raises:
            ValueError if adsorbate splitting failed
        """

        # get ads1 and ads2 from ads using graph theory
        mapped_dis_idx = [self.ads_idx.index(self.diss_idx[0]), self.ads_idx.index(self.diss_idx[1])]
        graph = self.adsorbate.graph(neighbour_distance=1.6)
        broken_graph = graph.delete_edges([tuple(mapped_dis_idx)])
        graph_components = broken_graph.components()

        # check if ads is split to 2 components
        if len(graph_components) != 2:
            raise ValueError("Adsorbate splitting failed")

        if mapped_dis_idx[0] in graph_components[1] and mapped_dis_idx[1] in graph_components[0]:
            ads1_idx, ads2_idx = graph_components[1], graph_components[0]
            ads1_new_bond = graph_components[1].index(mapped_dis_idx[0])
            ads2_new_bond = graph_components[0].index(mapped_dis_idx[1])

        elif mapped_dis_idx[0] in graph_components[0] and mapped_dis_idx[1] in graph_components[1]:
            ads1_idx, ads2_idx = graph_components[0], graph_components[1]
            ads1_new_bond = graph_components[0].index(mapped_dis_idx[0])
            ads2_new_bond = graph_components[1].index(mapped_dis_idx[1])
        else:
            raise ValueError("Adsorbate splitting failed")

        ads1 = self.adsorbate[ads1_idx].copy()
        ads2 = self.adsorbate[ads2_idx].copy()

        # get denticity of ads1 and ads2 using graph theory
        from catplat.structure.adsorbate import Adsorbate
        ads1_bond = Adsorbate(atoms=ads1, connectivity=self.ads1_connectivity, avg_coord_num=self.ads1_acn).get_bonds()
        ads2_bond = Adsorbate(atoms=ads2, connectivity=self.ads2_connectivity, avg_coord_num=self.ads2_acn).get_bonds()

        if len(ads1_bond) == 0:
            ads1_bond = Adsorbate(atoms=ads1, connectivity=self.ads1_connectivity,
                                  avg_coord_num=self.ads1_acn).get_bonds(length=1)
        if len(ads2_bond) == 0:
            ads2_bond = Adsorbate(atoms=ads2, connectivity=self.ads2_connectivity,
                                  avg_coord_num=self.ads2_acn).get_bonds(length=1)

        # adsorb ads1 monodentate
        # get site idx of ads1 and ads2
        ads1_site, ads2_site = self._get_ads_site(ads1=ads1, ads1_bond=ads1_bond, ads1_new_bond=ads1_new_bond,
                                                  ads2=ads2, ads2_bond=ads2_bond, ads2_new_bond=ads2_new_bond)

        # create unmapped final_state structure
        final_state_unmapped = self.slab.copy()

        # get structure of bonded ads1
        if self.ads1_bind:
            ads1_bound = self._bind_ads(ads=ads1, ads_bond=ads1_bond, ads_site=ads1_site)
            final_state_unmapped += ads1_bound
        else:
            final_state_unmapped += ads1

        # get structure of bonded ads1
        if self.ads2_bind:
            ads2_bound = self._bind_ads(ads=ads2, ads_bond=ads2_bond, ads_site=ads2_site)
            final_state_unmapped += ads2_bound
        else:
            final_state_unmapped += ads2

        # map ads_new back to ads
        ads_map = ads1_idx + ads2_idx

        ads_map_order = list(range(len(ads_map)))
        ads_map, ads_map_order = zip(*sorted(zip(ads_map, ads_map_order)))

        ads_new = final_state_unmapped[len(self.slab):]  # remove slab from complex
        ads_new_remap = Atoms()
        for i in ads_map_order:
            ads_new_remap += ads_new[i]  # remap atom index to match self.ads

        # map ads_new + slab back to initial input
        fs_temp = self.slab + ads_new_remap
        fs_map = self.slab_idx + self.ads_idx
        fs_map_order = list(range(len(fs_map)))
        fs_map, fs_map_order = zip(*sorted(zip(fs_map, fs_map_order)))

        final_state = self.initial.copy()
        del final_state[:]

        for i in fs_map_order:
            # remap atom index to match self.initial_state
            final_state += fs_temp[i]

        # set fixed atoms
        existing_fixed_atoms = self.initial.fixed_atoms_indices()
        final_state = final_state.fix_atoms(existing_fixed_atoms)

        return final_state

    def _get_ads_site(self, ads1: AtomsWrapper, ads1_bond: list, ads1_new_bond: int, ads2: AtomsWrapper,
                      ads2_bond: list, ads2_new_bond: int):
        """ Get adsorption site idx for ads1 and ads2

        Args:
            ads1_bond, ads2_bond (list): Bonds for ads1 and ads2 resepctively.
            ads1, ads2 (AtomsWrapper): Structure of ads1 and ads2 respectively
            ads1_new_bond, ads2_new_bond (int): Newly formed unsaturated atom index after ads1 dissociates from ads2

        Returns:
            Adsorption site for ads1 ands ads2. Int if monodentate and list of len 2 if bidentate.

        Raises:
            ValueError if len(ads1_bond) or len(ads2_bond) is not 1 or 2.

        """

        if len(ads1_bond) == 1 and len(ads2_bond) == 1:
            return self._mono_mono(ads1=ads1, ads1_bond=ads1_bond, ads2=ads2, ads2_bond=ads2_bond)
        elif len(ads1_bond) == 2 and len(ads2_bond) == 2:
            return self._bi_bi(ads1=ads1, ads1_bond=ads1_bond, ads1_new_bond=ads1_new_bond, ads2=ads2,
                               ads2_bond=ads2_bond, ads2_new_bond=ads2_new_bond)
        elif len(ads2_bond) == 2 and len(ads1_bond) == 1:
            return self._mono_bi(ads1=ads1, ads1_bond=ads1_bond, ads2=ads2, ads2_bond=ads2_bond,
                                 ads2_new_bond=ads2_new_bond)
        elif len(ads1_bond) == 2 and len(ads2_bond) == 1:
            return self._bi_mono(ads1=ads1, ads1_bond=ads1_bond, ads1_new_bond=ads1_new_bond, ads2=ads2,
                                 ads2_bond=ads2_bond)
        else:
            raise ValueError("Error getting site idx for ads1/ads2")

    def _mono_mono(self, ads1, ads1_bond, ads2, ads2_bond):
        """ Get sites where ads1 and ads2 are both monodentate """
        ads1_site = self._filter_sites(point=ads1[ads1_bond[0]].position, args_preset='ads1 monodentate', n=0)
        ads2_site = self._filter_sites(point=ads2[ads2_bond[0]].position, args_preset='ads2 monodentate', n=0)

        # check if ads2_site is the same as ads1_site
        if ads2_site == ads1_site:
            ads2_site = self._filter_sites(point=ads2[ads2_bond[0]].position, args_preset='ads2 monodentate', n=1)
        return ads1_site, ads2_site

    def _bi_bi(self, ads1, ads1_bond, ads1_new_bond, ads2, ads2_bond, ads2_new_bond):
        """ Get sites where ads1 and ads2 are both bidentate. 
        NOTE: DOUBLE BIDENTATE HAVE NOT BEEN FULLY TESTED.

        Raises:
         ValueError: If ads1 existing site is the same as ads2 existing site
        """

        ads1_existing_bond = [i for i in range(len(ads1_bond)) if ads1_bond[i] != ads1_new_bond][0]
        ads2_existing_bond = [i for i in range(len(ads2_bond)) if ads2_bond[i] != ads2_new_bond][0]

        ads1_existing_site = self._filter_sites(point=ads1[ads1_existing_bond].position, args_preset='closest site',
                                                n=0)
        ads2_existing_site = self._filter_sites(point=ads2[ads2_existing_bond].position, args_preset='closest site',
                                                n=0)

        if ads1_existing_site == ads2_existing_site:
            raise ValueError('Structure error!')

        # check if ads1_new_site is the same as ads1_existing_site or ads2_existing_site
        for n in range(0, 4):
            ads1_new_site = self._filter_sites(point=ads1[ads1_new_bond].position, args_preset='ads1 bidentate', n=n)
            if (ads1_new_site == ads1_existing_site) or (ads1_new_site == ads2_existing_site):
                pass
            else:
                break

        # check if ads2_new_site is the same as ads1_existing_site, ads2_existing_site or ads1_new_site
        for n in range(0, 4):
            ads2_new_site = self._filter_sites(point=ads2[ads2_new_bond].position, args_preset='ads2 bidentate', n=n)
            if (ads2_new_site == ads1_existing_site) or (ads2_new_site == ads2_existing_site) or (
                    ads2_new_site == ads1_new_site):
                pass
            else:
                break

        ads1_site, ads2_site = [0, 0], [0, 0]
        ads1_site[ads1_bond.index(ads1_existing_bond)] = ads1_existing_site
        ads1_site[ads1_bond.index(ads1_new_bond)] = ads1_new_site

        ads2_site[ads2_bond.index(ads2_existing_bond)] = ads2_existing_site
        ads2_site[ads2_bond.index(ads2_new_bond)] = ads2_new_site

        return ads1_site, ads2_site

    def _mono_bi(self, ads1, ads1_bond, ads2, ads2_bond, ads2_new_bond):
        """ Get sites where ads1 is monodentate while ads2 is bidentate """
        ads2_existing_bond = [i for i in range(len(ads2_bond)) if ads2_bond[i] != ads2_new_bond][0]
        ads2_existing_site = self._filter_sites(point=ads2[ads2_existing_bond].position, args_preset='closest site',
                                                n=0)

        # check if ads2_new_site is the same as ads2_existing_site
        for n in range(0, 4):
            ads2_new_site = self._filter_sites(point=ads2[ads2_new_bond].position, args_preset='ads2 bidentate', n=n)
            if (ads2_new_site == ads2_existing_site):
                pass
            else:
                break

        ads2_site = [0, 0]
        ads2_site[ads2_bond.index(ads2_existing_bond)] = ads2_existing_site
        ads2_site[ads2_bond.index(ads2_new_bond)] = ads2_new_site

        # check if ads1_site is the same as ads2_existing_site or ads2_new_site
        for n in range(0, 4):
            ads1_site = self._filter_sites(point=ads1[ads1_bond[0]].position, args_preset='ads1 monodentate', n=n)
            if (ads1_site == ads2_existing_site) or (ads1_site == ads2_new_site):
                pass
            else:
                break
        return ads1_site, ads2_site

    def _bi_mono(self, ads1, ads1_bond, ads1_new_bond, ads2, ads2_bond):
        """ Get sites where ads1 is bidentate while ads2 is monodentate """
        if len(ads1_bond) == 2 and len(ads2_bond) == 1:
            ads1_existing_bond = [i for i in range(len(ads1_bond)) if ads1_bond[i] != ads1_new_bond][0]
            ads1_existing_site = self._filter_sites(point=ads1[ads1_existing_bond].position, args_preset='closest site',
                                                    n=0)

            # check if ads1_new_site is the same as ads1_existing_site
            for n in range(0, 4):
                ads1_new_site = self._filter_sites(point=ads1[ads1_new_bond].position, args_preset='ads1 bidentate',
                                                   n=n)
                if (ads1_new_site == ads1_existing_site):
                    pass
                else:
                    break

            ads1_site = [0, 0]
            ads1_site[ads1_bond.index(ads1_existing_bond)] = ads1_existing_site
            ads1_site[ads1_bond.index(ads1_new_bond)] = ads1_new_site

            # check if ads2_site is the same as ads1_existing_site or ads1_new_site
            for n in range(0, 4):
                ads2_site = self._filter_sites(point=ads2[ads2_bond[0]].position, args_preset='ads2 monodentate', n=n)
                if (ads2_site == ads1_existing_site) or (ads2_site == ads1_new_site):
                    pass
                else:
                    break
            return ads1_site, ads2_site

    def _bind_ads(self, ads: AtomsWrapper, ads_bond: list, ads_site: list) -> AtomsWrapper:
        """ Binds adsorbate to slab. 

        Args: 
            ads (AtomsWrapper): Structure of adsorbate
            ads_bond (list): Bond of adsorbate
            ads_site (list): Site for adsorbate

        Returns:
            Atomswrapper: Structure of adsorbate after binding to slab

        Raises:
            ValueError: If len(ads_bond) != 1 or 2

        """
        # monodentate adsorption
        if len(ads_bond) == 1:
            if len(ads) == 1 or len(ads) == 2:
                # no rotation required for adsorbates of len 1 and len 2
                return self._bind_monodentate(ads=ads, ads_bond=ads_bond, ads_site=ads_site, rotation=0)
            else:
                ads_list = [self._bind_monodentate(ads=ads, ads_bond=ads_bond, ads_site=ads_site, rotation=i) for i in
                            range(0, 360)]
                image_dist = [ads.image_distance(i) for i in ads_list]

                image_dist, ads_list = zip(*sorted(zip(image_dist, ads_list), key=lambda pair: pair[0]))
                return ads_list[0]

        elif len(ads_bond) == 2:
            return self._bind_bidentate(ads=ads, ads_bond=ads_bond, ads_site=ads_site)

        else:
            raise ValueError("More than 2 bonds detected! Only monodentate/bidentate binding is allowed")

    def _bind_monodentate(self, ads, ads_bond, ads_site, rotation):
        """ Returns the adsorbate after monodentate binding to the slab. """
        builder = ComplexMonodentateBuilder(slab=self.slab, adsorbate=ads, bonds=ads_bond, rotation=rotation)
        complex = builder.adsorb_monodentate_on_site(idx=ads_site)
        complex_atoms = AtomsWrapper.from_atoms(complex[0].atoms)
        adsorbate = complex_atoms[len(self.slab):]
        return adsorbate

    def _bind_bidentate(self, ads, ads_bond, ads_site):
        """ Returns the adsorbate structure after bidentate binding to the slab. """
        builder = ComplexBidentateBuilder(slab=self.slab, adsorbate=ads, bonds=ads_bond)
        complex = builder.adsorb_bidentate_on_edge(edge=ads_site)
        complex_atoms = AtomsWrapper.from_atoms(complex.atoms)
        adsorbate = complex_atoms[len(self.slab):]
        return adsorbate

    def _get_ts(self, initial_state, final_state):
        """Derives the transition state guess structure using pyatoms.io.ase.neb.NEBCreator. Atoms are interpolated with 'idpp'. 
        Difference in bond distance between TS and IS and can be set with self.zoom and self.ts_tol (default=0.5)

        Args:
            initial_state (AtomsWrapper): Structure of initial state.
            final_state (AtomsWrapper): Structure of final state.

        Returns:
            if self.zoom:
                AtomsWrapper: Structure of initial state, AtomsWrapper: Structure of transition state, AtomsWrapper: Structure of final state

            if not self.zoom:
                AtomsWrapper: Structure of transition state

        """

        creator = NEBCreator.from_initial_and_final_images(initial=initial_state,
                                                           final=final_state,
                                                           nimages=30,
                                                           interpolation='idpp')

        neb = creator.interpolate()

        bond_length = [i.get_distance(self.diss_idx[0], self.diss_idx[1]) for i in neb.images]

        if self.zoom:
            is_index = np.argwhere(bond_length - bond_length[0] > self.ts_tol - self.zoom_tol)[0][0]
            ts_index = np.argwhere(bond_length - bond_length[0] > self.ts_tol)[0][0]

            try:
                fs_index = np.argwhere(bond_length - bond_length[0] > self.ts_tol + self.zoom_tol)[0][0]
            except:
                fs_index = -1  # use final image if fs distance too short

            initial_state, transition_state, final_state = neb.images[is_index], neb.images[ts_index], neb.images[
                fs_index]
            return initial_state, transition_state, final_state

        ts_index = ts_index = np.argwhere(bond_length - bond_length[0] > self.ts_tol)[0][0]
        transition_state = neb.images[ts_index]

        return transition_state
