class SlabBuilder(object):
    """ Heavily modified version of catkit.gen.surface.SlabGenerator.
    """

    def __init__(self,
                 bulk,
                 miller_index: Union[list, tuple, None] = None,
                 layers: Union[int, None] = None,
                 vacuum: Union[str, int, float, None] = None,
                 fixed_layers: Union[int, None] = None,
                 fixed_thickness: Union[str, int, float, None] = None,
                 size: Union[tuple[int, int], None] = None,
                 overlayer: Union[str, None] = None,
                 conventional: Union[bool, None] = None,
                 thickness: Union[str, int, float, None] = None,
                 x_length: Union[str, int, float, None] = None,
                 y_length: Union[str, int, float, None] = None,
                 slab_generator_tol: float = SLAB_GENERATOR_TOL):

        if (x_length is None and y_length is not None) or (y_length is None and x_length is not None):
            raise ValueError(f'Both x_length and y_length must be specified.')

        def preprocess(attribute):
            """ Preprocess a comparator str input to it's numerical value."""
            if isinstance(attribute, (int, float)):
                return attribute

            if isinstance(attribute, str):
                comp = Comparator.new(attribute)
                if comp._is_strictly_equals:
                    return comp.min_val
                if comp.max_val is not None:
                    raise ValueError("SlabBuilder can only handle '>' comparators.")
                if comp.min_val is not None:
                    return comp.min_val
                assert False

        x_length = preprocess(x_length)
        y_length = preprocess(y_length)
        thickness = preprocess(thickness)
        fixed_thickness = preprocess(fixed_thickness)
        vacuum = preprocess(vacuum)

        if size is None and x_length is None:
            size = UNITCELL_SIZE

        if layers is None and thickness is None:
            layers = NUM_LAYERS

        if fixed_layers is None and fixed_thickness is None:
            fixed_layers = NUM_FIXED_LAYERS

        if vacuum is None:
            vacuum = VACUUM

        if miller_index is None:
            miller_index = MILLER_INDEX

        if conventional is None:
            conventional = CONVENTIONAL

        if overlayer is None:
            overlayer = OVERLAYER

        if size and x_length:
            raise AttributeError(f'Unitcell_size and length cannot be defined together.')

        if layers and thickness:
            raise AttributeError(f'Layers and slab_thickness cannot be defined together.')

        if fixed_layers and fixed_thickness:
            raise AttributeError(f'fixed_layers and fixed_thickness cannot be defined together.')

        if not isinstance(miller_index, (tuple, list)):
            raise ValueError(f'Miller index must be list or tuple, instead was: {miller_index}')

        if not isinstance(conventional, bool):
            raise ValueError(f'conventional must be either True or False, instead was {conventional}')

        if x_length is None and y_length is None:
            self.slab_length = None
        else:
            self.slab_length = (x_length, y_length)

        self.slab_thickness: float = thickness
        self.layers: int = layers
        self.vacuum: int = vacuum
        self.fixed_layers: int = fixed_layers
        self.fixed_thickness: float = fixed_thickness
        self.tol: float = slab_generator_tol
        self.overlayer: typing.Union[str, None] = overlayer
        self.conventional: bool = conventional
        self.size: typing.Union[list, tuple] = size

        # miller index preprocessing
        miller_index = np.array(miller_index)
        if len(miller_index) == 4:
            miller_index[[0, 1]] -= miller_index[2]
            miller_index = np.delete(miller_index, 2)
        miller_index = (miller_index / list_gcd(miller_index)).astype(int)

        self.miller_index = miller_index
        self.bulk = bulk

    @cached_property
    def _bulk(self):
        """  Transformed bulk structure ready for cleaving. """
        oriented_bulk = self._orientate_unit_cell_to_miller_index(self.bulk, self.miller_index)
        return self._align_crystal(oriented_bulk)

    @property
    def unique_terminations(self) -> int:
        """ Determines the number of termination to test.

        Changelog:
            1. Replaced catkit function wtih get_unique_coordinates_cartesian_tol
        """
        return get_unique_coordinates_cartesian_tol(self._bulk)

    def generate_surfaces(self) -> list:
        """ Main function that creates the slab. """
        slabs = [self._create_slab(iterm) for iterm in range(len(self.unique_terminations))]
        slabs = self._unique_slabs(slabs)  # filter out unique slabs using symmetry filter
        logger.info(f'{len(slabs)} unique surfaces generated')
        return slabs



    def _orientate_unit_cell_to_miller_index(self, bulk, miller_index):
        """
        Copied from pymatgen.core.surface.SlabGenerator
        """

        def convert_miller_index(miller_index, atoms1, atoms2):
            """Return a converted miller index between two atoms objects."""
            from catkit.gen import utils
            recip1 = utils.get_reciprocal_vectors(atoms1)
            recip2 = utils.get_reciprocal_vectors(atoms2)

            converted_index = np.dot(miller_index, np.dot(recip1, np.linalg.inv(recip2)))
            converted_index = np.round(converted_index)
            converted_index = (converted_index / list_gcd(converted_index)).astype(int)
            return converted_index

        standard_bulk = symmetry.get_standardized_cell(bulk, tol=self.tol)
        primitive_bulk = symmetry.get_standardized_cell(bulk, primitive=True, tol=self.tol)

        # Add Wyckoff symbols of the bulk, will help with identfying types of sites in the slab system
        miller_index = (miller_index / list_gcd(miller_index)).astype(int)
        miller_index = convert_miller_index(miller_index, standard_bulk, primitive_bulk)

        initial_structure = AtomsWrapper.from_atoms(primitive_bulk).to_structure()

        sg = SpacegroupAnalyzer(initial_structure)
        initial_structure.add_site_property("bulk_wyckoff", sg.get_symmetry_dataset()["wyckoffs"])
        initial_structure.add_site_property("bulk_equivalent", sg.get_symmetry_dataset()["equivalent_atoms"].tolist())
        latt = initial_structure.lattice
        miller_index = reduce_vector(miller_index)

        # Calculate the surface normal using the reciprocal lattice vector.
        recp = latt.reciprocal_lattice_crystallographic
        normal = recp.get_cartesian_coords(miller_index)
        normal /= np.linalg.norm(normal)

        slab_scale_factor = []
        non_orth_ind = []
        eye = np.eye(3, dtype=int)
        for i, j in enumerate(miller_index):
            if j == 0:
                # Lattice vector is perpendicular to surface normal, i.e.,
                # in plane of surface. We will simply choose this lattice
                # vector as one of the basis vectors.
                slab_scale_factor.append(eye[i])
            else:
                # Calculate projection of lattice vector onto surface normal.
                d = abs(np.dot(normal, latt.matrix[i])) / latt.abc[i]
                non_orth_ind.append((i, d))

        # We want the vector that has maximum magnitude in the
        # direction of the surface normal as the c-direction.
        # Results in a more "orthogonal" unit cell.
        c_index, dist = max(non_orth_ind, key=lambda t: t[1])

        if len(non_orth_ind) > 1:
            lcm_miller = lcm(*[miller_index[i] for i, d in non_orth_ind])
            for (i, di), (j, dj) in itertools.combinations(non_orth_ind, 2):
                l = [0, 0, 0]
                l[i] = -int(round(lcm_miller / miller_index[i]))
                l[j] = int(round(lcm_miller / miller_index[j]))
                slab_scale_factor.append(l)
                if len(slab_scale_factor) == 2:
                    break

        slab_scale_factor.append(eye[c_index])
        slab_scale_factor = np.array(slab_scale_factor)

        # Let's make sure we have a left-handed crystallographic system
        if np.linalg.det(slab_scale_factor) < 0:
            slab_scale_factor *= -1

        # Make sure the slab_scale_factor is reduced to avoid unnecessarily large slabs
        reduced_scale_factor = [reduce_vector(v) for v in slab_scale_factor]
        slab_scale_factor = np.array(reduced_scale_factor)

        single = initial_structure.copy()
        single.make_supercell(slab_scale_factor)

        # When getting the OUC, lets return the most reduced
        # structure as possible to reduce calculations
        oriented_unit_cell = Structure.from_sites(single, to_unit_cell=True)
        oriented_unit_cell = AtomsWrapper.from_structure(oriented_unit_cell)
        oriented_unit_cell = GratomsWrapper.from_atoms(oriented_unit_cell)

        # compact unitcell
        oriented_unit_cell = compact_bulk_in_unitcell(oriented_unit_cell)
        return oriented_unit_cell

    def _align_crystal(self, bulk):
        """
        Copy from catkit.gen.surface.SlabGenerator
        Changelog:
        1) Simplified function to reduce dependency from ab basis vector alignment
        """
        import ase.build

        new_bulk = bulk

        # Align the longest of the ab basis vectors with x
        d = np.linalg.norm(new_bulk.cell[:2], axis=1)

        a = new_bulk.cell[0]
        a3 = np.cross(a, new_bulk.cell[1]) / np.max(d)
        ase.build.rotate(new_bulk, a3, (0, 0, 1), a, (1, 0, 0))

        # Ensure the remaining basis vectors are positive in their
        # corresponding axis
        for i in range(1, 3):
            if new_bulk.cell[i][i] < 0:
                new_bulk.cell[i] *= -1
        new_bulk.wrap(eps=1e-3)
        return new_bulk

    def get_slab_basis(self, iterm=0):
        """
        Copy from catkit.gen.surface.SlabGenerator

        Changelog:
        1. Assumes that slab_basis == [None] * len(terminations)
        2. Replace utils.get_unique_coordinates() with get_unique_coordinates_cartesian_tol()
        3. Remove the need to ibasis.set_cell(newcell) and ibasis.wrap() to prevent slab from generating inaccurate structures due to surface roughness.
        4. Added tolerance for slab cleaving during termination identification to ensure that the whole surface is cleaved.
        """

        def _get_num_layers(atoms, scaled_positions, t, tol=0.05):
            # note tol = 0.1 still have roughness
            scaled_positions = scaled_positions.copy()
            atoms = atoms.copy()

            scaled_positions[:, 2] += t
            atoms.set_scaled_positions(scaled_positions)
            atoms.wrap(pbc=True)
            return len(get_unique_coordinates_cartesian_tol(atoms, tol=tol))

        terminations = self.unique_terminations

        _basis = self._bulk.copy()
        ibasis = _basis.copy()

        zshift = terminations[iterm]
        scaled_positions = ibasis.get_scaled_positions()
        scaled_positions[:, 2] -= zshift

        translations = np.linspace(-self.tol, self.tol, 100)
        num_layers = [_get_num_layers(atoms=ibasis, scaled_positions=scaled_positions, t=t) for t in translations]

        best_translation = translations[np.argmin(num_layers)]
        scaled_positions[:, 2] += best_translation

        ibasis.set_scaled_positions(scaled_positions)
        ibasis.wrap(pbc=True)

        minimum_repetitions = self._get_minimum_repetitions(_basis)
        ibasis *= (1, 1, int(minimum_repetitions))

        get_unique_coordinates_cartesian_tol(ibasis)

        return ibasis



    def _get_minimum_repetitions(self, basis):
        bulk_layers = get_unique_coordinates_cartesian_tol(basis)
        layers_per_repetion = len(bulk_layers)

        if not self.slab_thickness:
            return np.ceil(self.layers / layers_per_repetion)

        minimum_repetitions = 1
        while True:
            trial_basis = basis.copy()
            trial_basis *= (1, 1, minimum_repetitions)  # repeat by 1 unit in the z-direction
            if max(trial_basis.positions[:, 2]) >= self.slab_thickness:
                break
            minimum_repetitions += 1  # increment
        return minimum_repetitions

    def get_slab(self, iterm=0):
        """
        Copy from catkit.gen.surface.SlabGenerator

        Changelog:
        1. Removed transform_ab step to prevent unexpected change in termination.
        2. Replace utils.get_unique_coordinates() with get_unique_coordinates_cartesian_tol()
        3. Assume layer_type to be 'trim'
        """

        slab = self.get_slab_basis(iterm).copy()

        # Orthogonalize the z-coordinate
        # Breaks bulk periodicity in the c-basis
        slab.cell[2] = [0, 0, slab.cell[2][2]]
        slab.set_pbc([1, 1, 0])

        # Trim the bottom of the cell, bulk symmetry may be lost
        # Assunes that layer_type is 'trim'
        zlayers = get_unique_coordinates_cartesian_tol(slab)
        reverse_sort = np.sort(zlayers)[::-1]
        ncut = reverse_sort[:self.layers][-1] * slab.cell[2][2]

        zpos = slab.positions[:, 2]
        index = np.arange(len(slab))
        del slab[index[zpos - ncut < -self.tol]]

        slab.cell[2][2] -= ncut
        slab.translate([0, 0, -ncut])

        tl = np.argmax(slab.get_scaled_positions()[:, 2])
        translation = slab[tl].position.copy()
        translation[2] = 0
        slab.translate(-translation)
        slab.wrap()

        return slab



    def _create_slab(self, termination_idx) -> Slab:
        """ Handles slab creation workflow. """
        slab = self.get_slab(iterm=termination_idx)  # generate (1,1) slab.
        slab = Slab(slab, miller_index=self.miller_index, provenance='autopy-slab')
        slab = slab.transform(vacuum=self.vacuum,
                              conventional=self.conventional,
                              fixed_layers=self.fixed_layers,
                              fixed_thickness=self.fixed_thickness,
                              size=self.size,
                              slab_length=self.slab_length,
                              overlayer=self.overlayer)
        return slab

    def _unique_slabs(self, slabs):
        """ SlabGenerator may generate repeated slabs, so here we try to find the unique slabs.
        Use grouper instead of e.g. set(slabs), since the order of the atoms in the generated slabs
        may be different even though their positions are exactly the same.
        """
        slab_atoms = [slab.atoms.copy() for slab in slabs]
        slab_layers = [slab.layers for slab in slabs]
        # StructuralSelfConsistentGrouper matches slabs that differ by rotation .e.g
        # Slab1:       Slab2:
        # Cu-Cu        Ag-Ag
        # Cu-Cu   ==   Cu-Cu
        # Ag-Ag        Cu-Cu

        # But we want these to be distinct slabs since only the top surface is exposed to adsorbates.
        # Here we use a hack to shift all first layer atoms by a small amount to break the rotational symmetry.
        # This hack would fail in the case of 2 layer slabs, so we check for that.
        # slabs = [slab.to_mutable() for slab in slabs]
        for idx, slab in enumerate(slab_atoms):
            slab.positions[np.argwhere(slab_layers[idx] == 1)] += [0, 0, 0.05]

        # Tested stol=2E-3 can differentiate between slabs that differ by just in one atom by 0.01  in one direction
        grouper = StructuralSelfConsistentGrouper.from_atoms(atoms=slab_atoms, stol=STRUCTURE_MATCHER_TOL_SLAB_BUILDER)
        _, idx_groups = grouper.group()

        # Take one representative structure from each group
        unique_slabs = [slabs[idx[0]] for idx in idx_groups]

        return unique_slabs
