    @classmethod
    def from_value(cls, value, **kwargs):
        """ Convenience method to instantiate AdsorbateAttribute subclass."""
        if value is None:
            raise ValueError('Value cannot be None')

        if isinstance(value, cls):
            return value

        if not isiterable(value, exclude_strings=True):
            value = (value,)

        if isinstance(value, np.ndarray):
            value = value.tolist()

        # for numpy.int32 and numpy.int64 types
        value = [int(i) if isinstance(i, (np.int32, np.int64)) else i for i in value]

        # for list of np.arrays
        value = [i.tolist() if isinstance(i, np.ndarray) else i for i in value]

        if not all([isinstance(i, cls.VALID_TYPE) for i in value]):
            raise TypeError(f"Invalid type {[type(i) for i in value]}. Must be {cls.VALID_TYPE}")

        return cls(value, **kwargs)

class AverageCoordinationNumber(AdsorbateAttribute):
    """ AverageCoordinationNumber is a tuple of value(s) that define the average coordination number
    of the site, corresponding to the bonded atom.

    Example:
        monodentate = (7,) | (7.0,) | ('<8',)
        bidentate = (7,7) | (7.0,7.0) | ('<8', '<8')
    """

    VALID_TYPE = (int, float, str)

    def __init__(self, value: Iterable, equal_tol=EQUAL_TOL['avg_coord_num']):
        self.equal_tol = equal_tol
        super().__init__(value=value)

    def __contains__(self, val) -> bool:
        """ Returns True if val matches any of the values in self.value. """
        return any([comp(val) for comp in self.comparators])

    @property
    def comparators(self) -> List[Comparator]:
        return [Comparator.new(value=v, equal_tol=self.equal_tol) for v in self.value]

    def match(self, avg_coord_num):
        if not isiterable(avg_coord_num, exclude_strings=True):
            avg_coord_num = (avg_coord_num,)

        if len(avg_coord_num) != self.denticity:
            raise ValueError(f'len(avg_coord_num) != len(self.value).')

        passed = [comp(val) for val, comp in zip(avg_coord_num, self.comparators)]
        return all(passed)

class Adsorbate(object):
    """ Properties that define an adsorbate """

    def __init__(self,
                 atoms: AtomsWrapper,
                 adsorption_sites=None,
                 bonds: Union[Bonds, None] = None,
                 sites: Union[Sites, None] = None,
                 adsorption_vector: Union[np.ndarray, None] = None,
                 rotation: Union[int, None] = None,
                 neighbour_distance: float = 1.6):

        if not isinstance(atoms, AtomsWrapper):
            atoms = AtomsWrapper.from_atoms(atoms)

        if isinstance(bonds, (str, type(None))):
            bonds = Bonds.from_adsorbate(adsorbate=atoms, denticity=bonds)

        if not isinstance(bonds, Bonds):
            bonds = Bonds.from_value(bonds)

        if len(bonds) > len(atoms):
            assert False, "Number of bonds cannot be greater than number of atoms."

        if rotation is None:
            rotation = 0

        if sites is not None:
            sites = Sites.from_value(sites)

        self.atoms = atoms
        self.bonds = bonds
        self._adsorption_sites = adsorption_sites
        self.neighbour_distance = neighbour_distance
        self.sites = sites
        self.rotation = rotation
        self.adsorption_vector = adsorption_vector

    def __len__(self):
        return len(self.atoms)

    def sites_info(self) -> Union[None, NamedTuple]:
        """ Site information of bonded site(s). """
        if self.sites is None:
            return None
        return self._adsorption_sites.get_sites_info(self.sites)

    @property
    def connectivity(self) -> Union[None, Connectivity]:
        """ Connectivity of the adsorbate."""
        if self.sites_info is None:
            return None
        return Connectivity.from_value(self.sites_info.connectivity)

    @property
    def avg_coord_num(self) -> Union[None, AverageCoordinationNumber]:
        """ Average coordination number of the adsorbate."""
        if self.sites_info is None:
            return None
        return AverageCoordinationNumber.from_value(self.sites_info.avg_coord_num)

    @property
    def adsorption_coords(self) -> np.ndarray:
        """ Adsorption_coords of the adsorbate. """
        return np.array([self.atoms[i].position for i in self.bonds])

        """ Denticity of the adsorbate. """
        return self.bonds.denticity

        return self.atoms.get_chemical_formula()

    @property

        """ Lewis structure of the adsorbate. """
        return LewisStructure(atoms=self.atoms)

    @property
    def _lewis_structure_no_multiple_bonds(self):
        """ Lewis structure of the adsorbate. Multiple bonds are not allowed. """
        return LewisStructure(atoms=self.atoms, multiple_bonds=False)

    @property
    def unpaired_electrons_single(self):
        """ Number of unsaturated bonds for each atom. Assumes each atom only can form single bonds.

        Example: CO with 3 unsaturated bonds on C and 1 unsaturated bond on O (C-O considered as single bond)
             >>> ads = Adsorbate.from_name('CO')
             >>> ads.unsaturated_bonds_single
             [3, 1]
        """
        return self._lewis_structure_no_multiple_bonds.num_unsaturated_bonds()

    @property
    def unpaired_electrons_multiple(self):
        """ Number of unsaturated bonds for each atom. Assumes atoms can form multiple bonds if 2
        unsaturated atoms are adjacent to each other.

        Example: CO with 0 unsaturated bonds on C and 2 unsaturated bond on O (CO considered as double bond)
             >>> ads = Adsorbate.from_name('CO')
             >>> ads.unsaturated_bonds_multiple
             [2, 0]
        """
        return self._lewis_structure.num_unsaturated_bonds()

    @property
    def lone_pairs(self):
        """ Number of lone pairs for each atom.

         Example 1: CO with 1 lone pair on C and 1 lone pair on O
             >>> ads = Adsorbate.from_name('CO')
             >>> ads.lone_pairs
             [1, 1]

         Example 2: CO2 with 0 lone pairs on C and 2 lone pairs on each O
             >>> ads = Adsorbate.from_name('CO2')
             >>> ads.lone_pairs
             [0, 2, 2]

    def radical_atoms(self):
        return self._lewis_structure.radical_atoms()

    def align(self, slab=None):
        """Aligns adsorbate to the y axis. If slab is specified, aligns the adsorbate to the y vector of the unit cell instead

        Args:
            slab (AtomsWrapper): Structure of the slab. Defaults to None.
        """

        adsorbate = self.atoms.copy()
        val, vec = adsorbate.get_moments_of_inertia(vectors=True)

        # Smallest moment of inertia corresponds to axis to which most
        # of the atoms are closest to, and thus the axis which most closely
        # defines the orientation of the adsorbate
        vec_idx = np.argmin(val)

        if slab is not None:
            cell = np.abs(slab.get_cell())
            y_cell = [i[1] for i in cell]
            y_idx = np.argmax(y_cell)
            adsorbate.rotate(vec[vec_idx], cell[y_idx])
            return adsorbate

        # align with y axis by default
        adsorbate.rotate(vec[vec_idx], 'y')

        return adsorbate

    def is_symmetrical_along_bond(self) -> bool:
        """ Returns True if the adsorbate is symetrical along bond vector. False otherwise. """

        if self.denticity != 2:
            raise ValueError('Adsorbate must have bond of len 2 to check symmetry')

        atom_positions = [self.atoms[i].position for i,
        _ in enumerate(self.bonds)]

        vec = atom_positions[0] - atom_positions[1]
        axis = [[i * vec for i in range(1, 6)], [i * -vec for i in range(1, 6)]]
        images = [Atoms('X5', positions=axis[i], pbc=(1, 1, 1), cell=[10, 10, 10]) +
                  self.atoms for i, _ in enumerate(self.bonds)]

        grouper = StructuralSelfConsistentGrouper.from_atoms(atoms=images)
        groups, _ = grouper.group()
        return len(groups) == 1
