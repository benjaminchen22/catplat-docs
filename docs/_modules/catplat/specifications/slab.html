    def _filter_terminations(self, slabs):
        if self.termination is None:
            return slabs

        filtered = [slab for slab in slabs if slab.termination == self.termination]
        return filtered

class SlabSpecificationsSet(SpecificationsSet,
                            metaclass=DictAttributesMetaclass,
                            other_attrs=SLAB_DEFAULT_VALUES,
                            types=SLAB_SPECS_TYPE_HINTS):
    """
    Main class containing the slab attributes provided by the user.

    Also contains some error checks for the attributes which are called in the QuerySet.

    Attributes:
        slab_atoms (list, tuple): Name of slab file in user's slab directory. (~/.autopy/slab or ~/autopy/autopy/data/slab).
                            When slab is specified, other bulk and slab attributes should not be specified.
        termination (list, tuple, None): Slab termination of alloy.
        overlayer (list, tuple, None): Replaces top layer atoms with overlayer atoms.
        miller_index (list, tuple, None): Miller index of slab to be created.
        unitcell_size (list, tuple, None): Size of slab to be created.
        num_layers (list, tuple, None): Number of layers of the slab.
        num_fixed_layers (list, tuple, None): Number of layers to be fixed.
        slab_fixed_thickness (float, None): Fixed atoms cutoff in Angstroms.
        vacuum (list, tuple, None): Amount of vacuum (in angstroms) to add on "each side" of the slab (i.e., total vacuum = vauum x 2) Vacuum should be an int.
        conventional (list, tuple, None): Toggle for creation of conventional slabs.
        slab_provenance (list, tuple, None): String for the origin of slab atoms. Can be ['autopy-slab'] or ['user-bulk']
        #TODO: Update documentation

    Examples:
        slab_set = SlabSpecificationsSet()
        slab_set.slab_atoms = ['user_slab']
        slab_set.termination = ['Cu']
        slab_set.overlayer = ['Cu']
        slab_set.miller_index = [(1,1,1)]
        slab_set.unitcell_size = [(2,2)]
        slab_set.num_layers = [4]
        slab_set.num_fixed_layers = [2]
        slab_set.slab_fixed_thickness = [2]
        slab_set.vacuum = [10]
        slab_set.conventional = [True]
        slab_set.slab_provenance = ['user-slab']
    """
    SPECIFICATIONS_CLASS = SlabSpecifications

    @property
    def is_user_slab(self):
        return self.slab_atoms != [None]

    def mutually_excl_slab_attrs_check(self):
        """ Check if mutually exclusive slab attrs are specified. """
        mutually_exclusive = ("overlayer", "miller_index", "unitcell_size", "num_layers",
                              "num_fixed_layers", "vacuum", "conventional")

        for v in mutually_exclusive:
            if self.__getattribute__(v) != [None]:
                raise ValueError("Slab attributes cannot be specified when user slab_atoms are specified.")

    def is_valid_user_slab_check(self):
        if not self.is_user_slab:
            return

        for i in self.slab_atoms:
            try:
                CatplatStructureBuilder.build_slab(i)
            except:
                raise ValueError(f"Slab atoms ({i}) cannot be read. Please ensure that the file is in "
                                 f"the slab directory ({CatplatUserSettings().user_slab_dir})")

    def generate_atoms(self, bulk_atoms=None, existing_atoms=None):
        slabs = [Slab(self.slab_atoms, provenance='user-slab')]
        slabs = self._filter_terminations(slabs)

        if existing_atoms is not None:
            slabs = [slab for slab in slabs if not self._is_existing(atoms=slab.atoms, existing_atoms=existing_atoms)]

        images = [slab.atoms for slab in slabs]
        properties = [{'termination': slab.termination,
                       'unitcell_size': slab.unitcell_size,
                       'num_layers': slab.num_layers,
                       'num_fixed_layers': slab.num_fixed_layers,
                       'vacuum': slab.vacuum,
                       'x_length': slab.x_length,
                       'y_length': slab.y_length,
                       'slab_thickness': slab.thickness,
                       'slab_chemsys': slab.chemsys,
                       'slab_formula': slab.chemical_formula,
                       'slab_provenance': slab.provenance,
                       'slab_acn': slab.surface_acn,
                       'slab_fixed_thickness': slab.fixed_thickness} for slab in slabs]
        return images, properties

    def generate_atoms(self, bulk_atoms=None, existing_atoms=None):
        """Get slab structure from SlabBuilder or read slab structure from user's slab directory."""

        builder = SlabBuilder(bulk=bulk_atoms,
                              miller_index=self.miller_index,
                              layers=self.num_layers,
                              overlayer=self.overlayer,
                              fixed_layers=self.num_fixed_layers,
                              fixed_thickness=self.slab_fixed_thickness,
                              size=self.unitcell_size,
                              x_length=self.x_length,
                              y_length=self.y_length,
                              thickness=self.slab_thickness,
                              vacuum=self.vacuum,
                              conventional=self.conventional)
        slabs = builder.generate_surfaces()  # returns a list of Slab

        if existing_atoms is not None:
            slabs = [slab for slab in slabs if not self._is_existing(atoms=slab.atoms, existing_atoms=existing_atoms)]

        slabs = self._filter_terminations(slabs)
        images = [slab.atoms for slab in slabs]
        properties = [{'termination': slab.termination,
                       'slab_provenance': slab.provenance,
                       'unitcell_size': slab.unitcell_size,
                       'num_layers': slab.num_layers,
                       'num_fixed_layers': slab.num_fixed_layers,
                       'conventional': slab.conventional,
                       'miller_index': slab.miller_index,
                       'vacuum': slab.vacuum,
                       'x_length': slab.x_length,
                       'y_length': slab.y_length,
                       'slab_thickness': slab.thickness,
                       'slab_chemsys': slab.chemsys,
                       'slab_formula': slab.chemical_formula,
                       'slab_acn': slab.surface_acn,
                       'slab_fixed_thickness': slab.fixed_thickness} for slab in slabs]
        return images, properties
