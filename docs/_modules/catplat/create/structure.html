class CatplatStructureBuilder(object):
    CATPLAT_ADSORBATE_DIR = os.path.join(os.path.dirname(__file__), '..', 'data', 'adsorbates')
    CATPLAT_SLAB_DIR = os.path.join(os.path.dirname(__file__), '..', 'data', 'slab')
    CATPLAT_BULK_DIR = os.path.join(os.path.dirname(__file__), '..', 'data', 'bulk')

    @classmethod
    def build_adsorbate(cls, adsorbate, user_dir=None, default_dir=None):
        """ Build an adsorbate structure from a name. """
        if user_dir is None:
            user_dir = CatplatUserSettings().user_adsorbate_dir

        if default_dir is None:
            default_dir = cls.CATPLAT_ADSORBATE_DIR

        return StructureBuilder(structure=adsorbate,
                                user_dir=user_dir,
                                default_dir=default_dir).build()
