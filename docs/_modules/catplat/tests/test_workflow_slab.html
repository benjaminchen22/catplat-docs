    @pytest.mark.parametrize('disable_optimization, completed', [(True, False), (False, True)])
    def test_ml_preoptimization(self, data_dir, blueprint_pt, catplat_test_project_settings, tmpdir,
                                fake_vasp_jobrunner, disable_optimization, completed):

        database = random_string(use_digits=False)
        with CatplatUserSettings.tmp_settings(CATPLAT_MLP_DATABASE=database) as settings:
            # Upload the MACE model to the database
            filename = os.path.join(data_dir, 'TestWorkflow', 'MACE-AgO.model')
            storage = PotentialsMySQLStorage.from_user_settings(settings=settings, database_name=database)
            storage.upload(filename=filename, mlip='MACE', elements=['Ag', 'O'])

            # Create and run the workflow
            qs = QuerySet()
            qs.bulk.chemsys = ['Ag']
            qs.slab.miller_index = [(1, 1, 1)]

            blueprint = WorkflowBlueprint.from_folder(query_set=qs, folder=tmpdir, ml_preoptimization=True)
            bulk_ag = bulk('Ag', 'fcc', a=4.09, cubic=True)

            workflow = SlabWorkflow.from_blueprint(blueprint=blueprint, bulk_atoms=bulk_ag,
                                                   project_settings=catplat_test_project_settings)[0]
            workflow.disable_optimization = disable_optimization
            workflow.run(jobrunner=fake_vasp_jobrunner)
            assert workflow.is_complete() == completed

        # Check the entries in the database
        database = workflow.database()

        # Cannot query by "chemsys" (which is not defined), must be "slab_chemsys"
        dft_query = Query(slab_chemsys='Ag', miller_index=(1, 1, 1), relaxed=True, ml_preoptimization=False)
        dft_entries = database.retrieve(query=dft_query)

        ml_query = Query(slab_chemsys='Ag', miller_index=(1, 1, 1), relaxed=True, ml_preoptimization=True)
        ml_entries = database.retrieve(query=ml_query)

        assert len(ml_entries) == 1

        if not disable_optimization:
            assert len(dft_entries) == 1
            assert dft_entries[0].ml_preoptimization_id == ml_entries[0].id

        # Test rerunning works even in different calculation folder
        blueprint.calculation_folder = tmpdir + '_rerun'
        os.mkdir(blueprint.calculation_folder)

        workflow = SlabWorkflow.from_blueprint(blueprint=blueprint, bulk_atoms=bulk_ag,
                                               project_settings=catplat_test_project_settings)[0]
        workflow.disable_optimization = False
        workflow.run(jobrunner=fake_vasp_jobrunner)
        assert workflow.is_complete()

        # Should not rerun ML preoptimization
        assert not os.path.exists(os.path.join(blueprint.calculation_folder, 'id1', 'ml_preopt'))
        dft_entries = database.retrieve(query=dft_query)
        ml_entries = database.retrieve(query=ml_query)

        assert len(ml_entries) == 1
        assert len(dft_entries) == 1
        assert dft_entries[0].ml_preoptimization_id == ml_entries[0].id

    def test_slab_to_adsorbate_magmoms(self, bulk_ni, tmpdir, catplat_test_project_settings, fake_vasp_jobrunner):
        qs = QuerySet()
        qs.bulk.chemsys = ['Ni']
        qs.slab.miller_index = [(1, 1, 1)]
        qs.slab.num_layers = [4]
        qs.slab.num_fixed_layers = [3]
        qs.adsorbate.adsorbate_atoms = ['H']

        blueprint = WorkflowBlueprint.from_folder(query_set=qs, folder=tmpdir)

        slab_workflow = SlabWorkflow.from_blueprint(blueprint=blueprint,
                                                    bulk_atoms=bulk_ni,
                                                    project_settings=catplat_test_project_settings)[0]

        slab_workflow.run(jobrunner=fake_vasp_jobrunner)

        incar_settings = catplat_test_project_settings.incar_settings()
        default_magmom_non_magnetic = incar_settings.default_magnetization
        default_magmom_Ni = incar_settings.magmom['Ni']

        slab_workflow_magmom = slab_workflow.optimization_job().steps[0].incar['MAGMOM']
        assert slab_workflow_magmom == [default_magmom_Ni] * 4

        adsorbate_workflow = slab_workflow.child_workflows[0]
        adsorbate_workflow_magmom = adsorbate_workflow.optimization_job().steps[0].incar['MAGMOM']
        assert adsorbate_workflow_magmom == [default_magmom_Ni] * 4 + [default_magmom_non_magnetic]

    def test_kpoints_consistency(self, pt_miller_index_workflow_completed):
        """ Test that the same kpoints are used for miller index and adsorbate workflows.
        Different kpoints could arise when using kpoints settings that depend on the structure
        e.g. from autogr density/reciprocal density, leading to slab and adsorbate results
        that cannot be compared -- we want to prevent that.
        """

        workflow = pt_miller_index_workflow_completed
        miller_index_job = workflow.optimization_job()
        miller_index_kpoints = miller_index_job.settings.kpoints_settings

        adsorbate_job = workflow.child_workflows[0].optimization_job()
        adsorbate_kpoints = adsorbate_job.settings.kpoints_settings

        for k1, k2 in zip(miller_index_kpoints, adsorbate_kpoints):
            assert k1 == k2

    def test_attributes_catplat_slab(self, blueprint_pt, bulk_pt, catplat_test_project_settings):
        """ Test slab attributes from catplat slab. """
        slab_specs = SlabSpecifications.from_attr(miller_index=(1, 1, 1))
        builder = SlabWorkflowBuilder(slab_specs=slab_specs, bulk_atoms=bulk_pt,
                                      project_settings=catplat_test_project_settings,
                                      blueprint=blueprint_pt)
        workflows = builder.build()
        assert len(workflows) == 1
        db = workflows[0].database()
        attributes = db.retrieve_all()[0].attributes

        assert attributes.relaxed is attributes.errored is attributes.ignored is False
        assert attributes.workflow_type == 'slab'
        assert attributes.version == catplat.__version__
        assert attributes.session_id is not None
        assert attributes.date is not None

        assert attributes.slab_atoms is None
        assert attributes.termination == 'Pt'
        assert attributes.miller_index == (1, 1, 1)
        assert attributes.overlayer is None
        assert attributes.num_layers == 4
        assert attributes.num_fixed_layers == 2
        assert pytest.approx(attributes.vacuum, 0.001) == 10
        assert attributes.conventional is True
        assert attributes.slab_provenance == 'autopy-slab'
        assert pytest.approx(attributes.x_length, 0.001) == 2.771
        assert pytest.approx(attributes.y_length, 0.001) == 2.771
        assert pytest.approx(attributes.slab_thickness, 0.001) == 6.789
        assert attributes.slab_chemsys == 'Pt'
        assert attributes.slab_formula == 'Pt4'
        assert attributes.slab_acn == 9
