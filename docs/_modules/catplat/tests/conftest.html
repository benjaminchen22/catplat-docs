@pytest.fixture(scope='module')
def test_project(test_project_path, fake_vasp_jobrunner,
                 bulk_job_twostep, surface_job_twostep, adsorbed_job_twostep, bulk_job_unstarted_twostep):
    # Create test project
    p = Project.from_folder(folder=test_project_path, jobrunner=fake_vasp_jobrunner, name='catplat-test')

    # Save project settings
    settings = p.settings()
    settings.save(config_dir=test_project_path)

    # Populate the test database
    data_bulk = VaspData.from_job(bulk_job_twostep)
    data_surface = VaspData.from_job(surface_job_twostep)
    data_adsorbed = VaspData.from_job(adsorbed_job_twostep)
    attributes_unrelaxed = DatabaseAttributes.from_dict(relaxed=False, errored=False)
    db = p.database()
    db.store(atoms=data_bulk.trajectory[0], attributes=attributes_unrelaxed)  # id1
    db.store(atoms=data_surface.trajectory[0], attributes=attributes_unrelaxed)  # id2
    db.store(atoms=data_adsorbed.trajectory[0], attributes=attributes_unrelaxed)  # id3
    db.store(atoms=bulk_job_unstarted_twostep.atoms, attributes=attributes_unrelaxed)  # id4

    db.store(atoms=data_bulk.trajectory[-1],
             attributes=DatabaseAttributes.from_dict(relaxed=True, errored=False, unrelaxed_id=1),
             data=data_bulk.to_dict())  # id5
    db.store(atoms=data_surface.trajectory[-1],
             attributes=DatabaseAttributes.from_dict(relaxed=True, errored=False, unrelaxed_id=2))  # id6
    db.store(atoms=data_adsorbed.trajectory[-1],
             attributes=DatabaseAttributes.from_dict(relaxed=False, errored=True, unrelaxed_id=3))  # id7
    db.store(atoms=data_adsorbed.trajectory[-1],
             attributes=DatabaseAttributes.from_dict(relaxed=True, errored=False, unrelaxed_id=3))  # id8
    db.store(atoms=data_adsorbed.trajectory[-1],
             attributes=DatabaseAttributes.from_dict(relaxed=True, errored=False, unrelaxed_id=3))  # id9

    return p
