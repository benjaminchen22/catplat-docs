    def test_project_test_mode(self, tmpdir, fake_vasp_jobrunner):
        """ In test mode, make sure that all calculations are not saved in the calculation path or in the database. """
        qs = QuerySet()
        qs.bulk.chemsys = ['Pt']
        qs.slab.miller_index = [(1, 0, 0)]
        qs.adsorbate.adsorbate_atoms = ['CO']
        qs.adsorbate.bonds = [[1]]
        qs.adsorbate.connectivity = [[1]]

        p = Project.from_folder(folder=tmpdir, jobrunner=fake_vasp_jobrunner, name=TEST_PROJECT_NAME)
        with p.test_mode() as ptest:
            results_bonds1 = ptest.retrieve(qs)

        assert len(results_bonds1) == 1
        # Binding through oxygen
        atoms = results_bonds1.entries[0].atoms
        c_idx = atoms.find_element('C')[0]
        o_idx = atoms.find_element('O')[0]
        assert atoms[c_idx].position[-1] > atoms[o_idx].position[-1]

        # Actual database should be still uncreated
        database_path = p.database_path
        assert not os.path.exists(database_path)

        # When calculating another system, should return new results. Previously had a bug where a test run wrote
        # into the calculation folder and then a subsequent test run retrieved these results which are inappropriate
        # for the query
        qs.adsorbate.bonds = [[0]]
        results_bonds2 = p.retrieve(qs)
        assert len(results_bonds2) == 1

        # Binding through carbon
        atoms = results_bonds2.entries[0].atoms
        c_idx = atoms.find_element('C')[0]
        o_idx = atoms.find_element('O')[0]
        assert atoms[c_idx].position[-1] < atoms[o_idx].position[-1]

    @pytest.mark.parametrize('bond_length_buffer,expected_result', [(0, 1), (2, 1), (10, 2)])
    def test_adsorbates_not_recalculated_if_binding_site_changes(self, tmpdir, fake_vasp_jobrunner, bond_length_buffer,
                                                                 expected_result):
        """ Test on 4-fold hollow site of 211 surface as it has a non-normal adsorption vector that will change the
         x-y adsorption coordinates if the bond length is changed.

         If using original buffer of 0, then we expect catplat to recognize that the new adsorbate generated during
         recalculation are the same as the old one, and only one result is produced.

         If using slightly different buffer of 2, then the new adsorption coords are slightly different from the old one
         (smaller than the specified tolerance), so it will NOT trigger a recalculation and only one result is produced.

         If using large new buffer of 10, then the new adsorption coords are very much different from the old one
         (greater than the specified tolerance), so it will trigger a recalculation and more results are produced.
         """
        qs = QuerySet()
        qs.bulk.chemsys = ['Pt']
        qs.slab.miller_index = [(2, 1, 1)]
        qs.adsorbate.adsorbate_atoms = ['H']
        qs.adsorbate.connectivity = [[4]]

        p = Project.from_folder(folder=tmpdir, jobrunner=fake_vasp_jobrunner, name=TEST_PROJECT_NAME)
        results = p.retrieve(qs)
        assert len(results) == 1

        # Test recalculation
        previous_buffer = catplat.settings.constants.DEFAULT_BOND_LENGTH_BUFFER
        try:
            catplat.settings.constants.DEFAULT_BOND_LENGTH_BUFFER = bond_length_buffer
            results = p.retrieve(qs)
            assert len(results) == expected_result
        finally:
            catplat.settings.constants.DEFAULT_BOND_LENGTH_BUFFER = previous_buffer

    def test_query_criteria(self, tmpdir, fake_vasp_jobrunner):
        """ Test for bug where session_id and version are being inherited by child workflows,
       leading to multiple relaxed=False structures to be written in the database. """

        p = Project.from_folder(folder=tmpdir, jobrunner=fake_vasp_jobrunner, name=TEST_PROJECT_NAME)

        def get_unrelaxed_structures_in_db(p):
            qs_unrelaxed = QuerySet()
            qs_unrelaxed.other.relaxed = [False]
            results_unrelaxed = p.retrieve(qs_unrelaxed)
            return len(results_unrelaxed)

        # Start off with bulk workflow
        qs = QuerySet()
        qs.bulk.mpid = ['mp-30']

        results = p.retrieve(qs)
        assert len(results) == 1  # 1 bulk
        # bulk relaxed == bulk unrelaxed == 1
        assert get_unrelaxed_structures_in_db(p) == 1

        # Start slab workflow with another session_id
        get_session_id(refresh=True)
        qs.slab.miller_index = [(1, 0, 0)]
        qs.slab.unitcell_size = [(3, 3)]

        results = p.retrieve(qs)
        assert len(results) == 1  # 1 slab
        assert get_unrelaxed_structures_in_db(p) == 2  # 1 bulk + 1 slab

        # Start adsorbate workflow with yet another session_id
        get_session_id(refresh=True)
        qs.adsorbate.adsorbate_atoms = ['H']

        results = p.retrieve(qs)
        assert len(results) == 3  # top, hollow, 4-fold complex
        assert get_unrelaxed_structures_in_db(p) == 5  # 1 bulk + 1 slab + 3 adsorbate

    def test_ignore_flag(self, tmpdir, fake_vasp_jobrunner):
        """ Test behaviour of ignored flag. """
        qs = QuerySet()
        qs.other.relaxed = [True, False]
        qs.bulk.chemsys = ['Cu']

        p = Project.from_folder(folder=tmpdir, jobrunner=fake_vasp_jobrunner, name=TEST_PROJECT_NAME)
        results = p.retrieve(qs)
        assert len(results) == 2  # id1(unrelaxed), id2(relaxed)

        # ignore calculated entry (id2),
        # new calculated entry should be generated (id3).
        from catplat.database.ase import Database
        db = Database(p.database_path)
        db.update(db_id=2, key='ignored', value=True)

        results = p.retrieve(qs)
        assert len(results) == 2  # id1 + id3
        assert len(db) == 3  # ids 1-3

        # ignore id1 and id2, only id3(relaxed) is not ignored
        # 2 new entries should be generated: id4(unrelaxed) and id5(relaxed)
        db.update(db_id=1, key='ignored', value=True)
        results = p.retrieve(qs)
        assert len(results) == 3  # id3 + id4 + id5
        assert len(db) == 5  # ids 1-5

        qs.other.ignored = [True]
        results = p.retrieve(qs)
        assert len(results) == 2  # id1 + id2 are ignored

    def test_slab_fixed_thickness(self, tmpdir, fake_vasp_jobrunner):

        qs = QuerySet()
