    def test_build_initial_magmoms(self, blueprint_pt, catplat_test_project_settings, request, pt111_3x3):
        """ Test that slab initial magmoms are preserved. """
        slab = pt111_3x3
        slab.set_initial_magnetic_moments(magmoms=[10] * len(slab))
        adsorbate_specs = AdsorbateSpecifications(adsorbate_atoms='H')
        builder = AdsorbateWorkflowBuilder(adsorbate_specs=adsorbate_specs, blueprint=blueprint_pt,
                                           project_settings=catplat_test_project_settings, surface_atoms=slab)

        workflows = builder.build()
        assert len(workflows) == 4

        database = workflows[0].database()
        entries = database.retrieve({'bonds': [0]})

        assert len(entries) == 4
        # Adsorbate default initial magmom is np.nan
        assert entries[0].atoms.get_initial_magnetic_moments() == pytest.approx([10] * len(slab) + [0])

    @pytest.mark.parametrize('adsorbate_atoms_and_magmoms', [('O2', [1, 1]), ('H', [0])])
    @pytest.mark.parametrize('slab', ['ni111', 'pt111'])
    def test_build_no_initial_magmoms(self, blueprint_ni, data_dir, catplat_test_project_settings, slab, ni111, pt111,
                                      adsorbate_atoms_and_magmoms):
        """
        Tests the outputted magmoms of the slab+ads complex.

        ni111 slab has no initial magmoms and so the lack of initial magmoms should be preserved as
        np.nans, whereas adsorbate magmoms (from ASE DB for example) should be preserved
        """
        adsorbate_atoms, ads_magmoms = adsorbate_atoms_and_magmoms

        if slab == 'ni111':
            surface_atoms = ni111
        elif slab == 'pt111':
            surface_atoms = pt111
        else:
            assert False

        adsorbate_specs = AdsorbateSpecifications(adsorbate_atoms=adsorbate_atoms)
        builder = AdsorbateWorkflowBuilder(adsorbate_specs=adsorbate_specs, blueprint=blueprint_ni,
                                           project_settings=catplat_test_project_settings, surface_atoms=surface_atoms)
        workflows = builder.build()
        atoms = workflows[0].db_entry.atoms

        # Check that the atoms have correct magmoms - both slab and adsorbate magmoms should be preserverd
        default_magmom = catplat_test_project_settings.incar_settings().default_magnetization
        assert pytest.approx(atoms.get_initial_magnetic_moments()) == [0] * len(surface_atoms) + ads_magmoms

        # Check that the incar is written properly
        job = workflows[0].optimization_job()
        incar_magmom = job.steps[0].incar.get('MAGMOM')
        if slab == 'pt111' and adsorbate_atoms == 'H':
            # All atoms are non magnetic
            assert incar_magmom is None
        elif slab == 'pt111' and adsorbate_atoms == 'O2':
            # O2 is magnetic, so treat system as magnetic
            assert incar_magmom == [default_magmom] * len(surface_atoms) + ads_magmoms
        elif slab == 'ni111' and adsorbate_atoms == 'H':
            # Ni111 is magnetic, so we treat the whole system as magnetic
            assert incar_magmom == [5.0] * len(surface_atoms) + [default_magmom]  # default magmom of H == 0.6
        elif slab == 'ni111' and adsorbate_atoms == 'O2':
            # Ni111 is magnetic, so we treat the whole system as magnetic
            assert incar_magmom == [5.0] * len(surface_atoms) + ads_magmoms

    def test_attributes_bidentate(self, blueprint_pt, pt111_3x3, catplat_test_project_settings):

        assert attributes.adsorbate_formula == 'H'
        assert attributes.bonds == [0]
        assert attributes.connectivity == [1]
