    def test_from_atomswrapper(self, data_dir):
        filepath = os.path.join(data_dir, 'POSCAR_AgO_p4x4')
        atoms = AtomsWrapper.from_file(filepath)
        magmoms = [10] * len(atoms)
        atoms.set_initial_magnetic_moments(magmoms)
        gratoms = GratomsWrapper.from_atoms(atoms)

        assert list(gratoms.get_initial_magnetic_moments()) == magmoms
        assert gratoms.to_atomswrapper() == atoms

    def test_translate(self, ago_p4x4_slab):
        gratoms = ago_p4x4_slab.copy()
        original_positions = gratoms.positions
        expected_positions = original_positions + np.array([0, 0, 10])

        actual_positions = gratoms.positions
        assert np.array_equal(expected_positions, actual_positions)

    def test_surface_atom(self, data_dir):
        """ Test that surface atom data is stored in"""

        def get_surface_atoms(slab):
            """Return surface atoms function from gratoms"""
            surf_atoms = np.where(slab.get_array('surface_atoms') > 0)[0]
            return surf_atoms

        slab_overlayer = GratomsWrapper.from_file(os.path.join(data_dir,"POSCAR_overlayer"))
        surface_atoms = np.array(slab_overlayer.find_element('Rh'))
        slab_overlayer.set_surface_atoms(surface_atoms)

        slab_atoms = slab_overlayer.to_atoms()

        # GratomsWrapper -> Atoms -> GratomsWrapper
        assert type(slab_atoms) == Atoms
        assert np.array_equal(get_surface_atoms(slab_atoms), surface_atoms)
        slab_gratomswrapper = GratomsWrapper.from_atoms(slab_atoms)
        assert np.array_equal(get_surface_atoms(slab_gratomswrapper), surface_atoms)
        assert np.array_equal(get_surface_atoms(slab_gratomswrapper._gratoms), surface_atoms)

        # GratomsWrapper -> AtomsWrapper -> GratomsWrapper
        slab_atomswrapper = slab_overlayer.to_atomswrapper()
        assert type(slab_atomswrapper) == AtomsWrapper
        assert np.array_equal(get_surface_atoms(slab_atomswrapper), surface_atoms)
        assert np.array_equal(get_surface_atoms(slab_atomswrapper._atoms), surface_atoms)
        slab_gratomswrapper = GratomsWrapper.from_atoms(slab_atomswrapper)
        assert np.array_equal(get_surface_atoms(slab_gratomswrapper), surface_atoms)
        assert np.array_equal(get_surface_atoms(slab_gratomswrapper._gratoms), surface_atoms)

        # Atoms --> AtomsWrapper
        slab_atomswrapper2 = AtomsWrapper.from_atoms(slab_atoms)
        assert np.array_equal(get_surface_atoms(slab_atomswrapper2), surface_atoms)
