    @pytest.mark.parametrize('attributes, expected',
                             (({'chemsys': 'Cu'}, ('Cu', 'Cu', 225, 0.0, 'mp-30', 'mp-bulk')),
                              ({'bulk_atoms': 'valid_bulk'}, ('Rh', 'Rh4', 225, None, None, 'user-bulk'))))
    def test_generate_atoms_bulkspecs(self, attributes, expected):
        """ Test generate_atoms() for BulkSpecs subclasses. """
        expected_properties = {'chemsys': expected[0], 'bulk_formula': expected[1], 'spacegroup': expected[2],
                               'e_above_hull': expected[3], 'mpid': expected[4], 'bulk_provenance': expected[5]}
        bulk_specs = BulkSpecifications.from_attr(**attributes)

        images, properties = bulk_specs.generate_atoms()
        properties = properties[0]

        assert len(images) == 1
        assert properties == expected_properties

    @pytest.mark.parametrize('e_above_hull, spacegroup, expected', (('<0.5', 225, 1), ('>=0', 194, 2), (0, None, 1)))
    def test_sieve_bulk(self, e_above_hull, spacegroup, expected):
        """ Test if specifications class is working correctly with QueryMPAPI. """
        attributes = {'chemsys': 'Cu', 'e_above_hull': e_above_hull, 'spacegroup': spacegroup}
        bulk_specs = MPBulkSpecifications(**attributes)
        images, properties = bulk_specs.generate_atoms()
        assert len(images) == len(properties) == expected
