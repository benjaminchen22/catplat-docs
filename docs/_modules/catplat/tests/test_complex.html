    def test_monodentate_attributes(self, pt111_3x3):
        builder = ComplexBuilder.from_adsorbate(slab=pt111_3x3, adsorbate='CO', bonds=[0])
        cmplx = builder.adsorb_monodentate_on_site(idx=40)
        cmplx.view(schematic=True)

        assert isinstance(cmplx, Complex)
        assert cmplx.bonds == [[0]]
        assert cmplx.sites == [[40]]
        assert cmplx.num_adsorbates == 1
        assert cmplx.adsorbate_indices == [[36, 37]]
        assert len(cmplx.adsorption_vector) == cmplx.num_adsorbates
        assert len(cmplx.adsorption_vector[0]) == 3
        assert cmplx.denticities == [1]
        assert cmplx.chemical_formula == 'COPt36'
        assert cmplx.slab_chemical_formula == 'Pt36'
        assert cmplx.adsorbates_chemical_formula == ['CO']
        assert cmplx.connectivities == [[1]]
        assert cmplx.avg_coord_num == [[9.0]]
        assert cmplx.bonds_mapped == [[36]]
        assert len(cmplx.adsorption_coords) == cmplx.num_adsorbates

    def test_multi_attributes(self, pt111_3x3):
        builder = ComplexBuilder.from_adsorbate(slab=pt111_3x3, adsorbate='CO', bonds=[0],
                                                quantity=2, sites_idx=[40, 41])
        cmplx = builder.build()[0]
        cmplx.view(schematic=True)

        assert isinstance(cmplx, Complex)
        assert cmplx.sites == [[40], [41]]
        assert cmplx.adsorbate_indices == [[36, 37], [38, 39]]
        assert cmplx.denticities == [1, 1]
        assert cmplx.chemical_formula == 'C2O2Pt36'
        assert cmplx.slab_chemical_formula == 'Pt36'
        assert cmplx.adsorbates_chemical_formula == ['CO', 'CO']
        assert cmplx.connectivities == [[1], [1]]
        assert cmplx.avg_coord_num == [[9.0], [9.0]]
        assert len(cmplx.adsorption_coords) == 2

    def test_is_similar_monodentate(self, pt111_3x3):
        adsorbate = 'CO'
        builder = ComplexBuilder.from_adsorbate(slab=pt111_3x3, adsorbate=adsorbate, bonds=[0])
        complex1 = builder.adsorb_monodentate_on_site(idx=0)
        complex2 = builder.adsorb_monodentate_on_site(idx=1)

        assert complex1.is_similar(image=complex1.atoms)
        assert complex2.is_similar(image=complex2.atoms)

        assert not complex1.is_similar(image=complex2.atoms)
        assert not complex2.is_similar(image=complex1.atoms)

    def test_min_distance_to_slab(self, pt111_3x3):
