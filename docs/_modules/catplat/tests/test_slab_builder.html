    @pytest.mark.parametrize('tol', [1E-3, 1E-2, 4E-2, 5E-2, 2E-1])
    def test_cu_bad_slab(self, data_dir, tol):
        """ This fcc bulk will give a weird-looking (111) slab if the tolerances are too strict.
        Tolerance of 5E-2 gives the normal slab (4E-2 does not work)."""
        filename = os.path.join(data_dir, 'POSCAR_Cu_bulk_badslab')
        bulk = AtomsWrapper.from_file(filename)
        builder = SlabBuilder(bulk=bulk, miller_index=(1, 1, 1), layers=6, size=(3, 3),
                              slab_generator_tol=tol)
        slabs = builder.generate_surfaces()
        slab = slabs[0]
        view(slabs)

        if tol in [5E-2, 2E-1]:  # test for valid cases
            assert slab.surface_acn == 9
            assert len(slabs) == 1  # only 1 slab generated.
            assert len(slab) == 6 * 9  # assert structure only have 6 layers * 9 atoms per layer.
            assert slab.num_layers == 6  # assert 6 layers.
        else:
            assert slab.surface_acn != 9

    @pytest.mark.parametrize(('miller, num_layers_for_surface'), (((2, 1, 1), 3), ((1, 1, 1), 1)))
    def test_overlayer(self, bulk_pt, miller, num_layers_for_surface):
        """ Test overlayer for 111 and 211 slabs. """
        builder = SlabBuilder(bulk=bulk_pt, overlayer='Rh', miller_index=miller, size=(2, 4), layers=12, fixed_layers=6)

        assert len(slabs) == 1

        slab = slabs[0]
        # view([slab])

        assert len(slab.find_element('Pt')) == (12 - num_layers_for_surface) * expected_atoms_per_layer
        assert slab.chemsys == 'Pt-Rh'
        assert slab.overlayer == 'Rh'

    def test_pt100(self, bulk_pt):
        """
        Simple test case for 100 surface.

        Integration test for the following parameters:
            SlabBuilder: miller_index, layers, fixed_layers, size, vacuum
            Slab: num_layers, unitcell_size, chemsys, termination, surface_acn, chemical_formula, vacuum

        """
        builder = SlabBuilder(bulk=bulk_pt, miller_index=(1, 0, 0), layers=5, size=(2, 2), fixed_layers=3, vacuum=10)
        slabs = builder.generate_surfaces()
        assert len(slabs) == 1
        slab = slabs[0]
        view(slabs)

        assert slab.num_layers == 5  # make sure its 5 layers
        assert len(slab) == 4 * 5  # 5 layers * 4 atoms
        assert len(slab.fixed_atoms_indices) == 4 * 3  # 3 fixed layers
        assert slab.unitcell_size == (2, 2)
        assert slab.num_fixed_layers == 3
        assert slab.chemsys == 'Pt'
        assert slab.termination == 'Pt'
        assert slab.surface_acn == 8
        assert slab.chemical_formula == 'Pt20'
        assert pytest.approx(slab.vacuum, 0.01) == 10

    @pytest.mark.parametrize('tol', [0.001, 0.01, 0.04, 0.05, 0.12])
    def test_au_slab(self, bulks_dir, tol):
        """ Too low tolerance will give bad slab"""
        bulk = AtomsWrapper.from_file(os.path.join(bulks_dir, 'POSCAR_Au_bulk_bad_slab'))
        slabs = SlabBuilder(bulk=bulk, miller_index=(1, 1, 1), size=(4, 4), slab_generator_tol=tol).generate_surfaces()
        assert len(slabs) == 1
        slab = slabs[0]
        # surface_atoms = slab.get_surface_atoms()

        if tol in [0.12]:  # test for valid cases
            # 4 x 4 111 surface as expected (16 atoms per layer)
            assert slab.surface_acn == 9
            assert slab.num_layers == 4  # assert 6 layers.
            assert len(slab) == 4 * 16
        elif tol in [0.001]:
            # Gives 111 surface but very large.. (32 atoms per layer)
            assert slab.surface_acn == 9
            assert slab.num_layers == 4
            assert len(slab) == 4 * 32
        else:
            # Bad slab
            assert slab.surface_acn != 9

    @pytest.mark.parametrize(('structure, miller, terminations'), (('POSCAR_CuPd_bulk_highkpoints', (0, 1, 3), 1),
                                                                   ('POSCAR_103', (1, 0, 3), 1),

    def test_cupd_rough(self, data_dir, structure, miller, terminations):
        """ Sample testing out of 137 bulks. These structures had roughness issues during code development.
        Assert that slab_generator_tol default is enough to generate nice CuPd 111 slabs."""
        filename = os.path.join(data_dir, 'CuPd-bulk-relaxed', structure)
        bulk = AtomsWrapper.from_file(filename)
        builder = SlabBuilder(bulk=bulk, miller_index=miller, size=[4, 1], layers=4)
        slabs = builder.generate_surfaces()

        layers = [slab.num_layers for slab in slabs]
        coordination = [slab.surface_acn for slab in slabs]

        assert len(set(layers)) == 1  # assert that all slab have the same number of layers
        assert all(i >= 4 for i in layers)  # assert that all slab have the same number of layers
        assert all(i == 9 for i in coordination)  # assert that 111 surfaces are generated
        assert len(slabs) == terminations  # assert that the correct number of slabs created

    def test_CuPd_slabs(self, data_dir, slab_number, miller, terminations):
        """ More robust test for full set of relaxed POSCARS"""

        # Note: POSCAR_7 may fail as the coordination is 9.68. The surface is (111)-like, but the bottom layers
        # are not perfectly aligned, causing some atoms to have 10 coordination.

        bulk = AtomsWrapper.from_file(data_dir + f'/CuPd-bulk-relaxed/POSCAR_{slab_number}')
        builder = SlabBuilder(bulk=bulk, miller_index=miller, size=(4, 4), layers=4)
        slabs = builder.generate_surfaces()
        view(slabs)

        coordination = [get_avg_coord_num(slab) for slab in slabs]
        assert len(slabs) == terminations
        assert set(coordination) == {9}
