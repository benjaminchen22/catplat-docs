def write_complexes(images, tmpdir):
    for idx, atoms in enumerate(images):
        folder = os.path.join(tmpdir, f'{idx}')
        os.mkdir(folder)
        poscar = os.path.join(folder, 'POSCAR')
        atoms.write(poscar, vasp5=True, direct=True)

    def test_avg_coord_num_comparator(self, rh112_3x2, rh112_coord_num_vs_sites):
        results1 = ComplexBuilder.from_adsorbate(slab=rh112_3x2, adsorbate='H',

        assert len(results3) == rh112_coord_num_vs_sites[9.5] + \
            rh112_coord_num_vs_sites[9.7] + rh112_coord_num_vs_sites[10.0]

    @pytest.mark.parametrize('slab, kwargs, expected', [
        ('pt211_2x3_conv', {'connectivity': [[1, 2]], 'avg_coord_num':[[7.0, 8.0]]}, 1),
        ('pt211_doped', {'connectivity': [[1, 2]], 'avg_coord_num':[[7.0, 8.0]]}, 2),
        ('pt211_2x3_conv', {'connectivity': [[1, 1]], 'avg_coord_num':[[7.0, 7.0]]}, 1),
        ('pt211_doped', {'connectivity': [[1, 1]], 'avg_coord_num':[[7.0, 7.0]]}, 2)])
    def test_reverse(self, slab, expected, kwargs, request):
        """
        For Pt211_doped slab:
        AB == BA for symmmetric adsorbates -> 1 structure.
        AB != BA for asymmetric adsorbates -> 2 structures.

        For normal Pt211 slab:
        AB == BA for both symmetric and asymmetric adsorbates -> 1 structure.

        """
        slab = request.getfixturevalue(slab)
        builder = ComplexBuilder.from_adsorbate(adsorbate='CO', slab=slab, bonds=[0, 1], **kwargs)
        complexes = builder.build()
        view(complexes)

        assert len(complexes) == expected
        assert all([list(cmplx.bonds[0]) == [0, 1] for cmplx in complexes])
        assert all([list(cmplx.connectivities[0]) == kwargs['connectivity'][0] for cmplx in complexes])
        assert all([list(cmplx.avg_coord_num[0]) == kwargs['avg_coord_num'][0] for cmplx in complexes])

    @pytest.mark.parametrize('bonds', (1, [1]))
    def test_bonds(self, cu100_4x4, bonds):
        """ Test that bonds=1 binds CO via C instead of O."""
        builder = ComplexBuilder.from_adsorbate(slab=cu100_4x4, adsorbate='CO', bonds=bonds)

        assert builder.bonds == [1]
        complexes = builder.build()

        for i in complexes:
            atoms = i.atoms
            c_idx = atoms.find_element('C')[0]
            o_idx = atoms.find_element('O')[0]
            # test that C z-pos is greater than O
            assert atoms[c_idx].position[2] > atoms[o_idx].position[2]

    def test_multi_integration(self, pt211_2x3_conv):
        """ Integration test for multiple arguments. """
        builder = ComplexBuilder.from_adsorbate(slab=pt211_2x3_conv, adsorbate='H', bonds=[0], quantity=2,
                                                avg_coord_num=[7], connectivity=[1], sites_idx=[84], distance_tol=0.8)
        complexes = builder.build()
        view(complexes)
        assert len(complexes) == 3

        complexes_sites = [cmplx.sites for cmplx in complexes]
        assert all([84 in sites for sites in complexes_sites])

        complexes_con = [cmplx.connectivities for cmplx in complexes]
        assert all([1 in con for con in complexes_con])

        complexes_acn = [cmplx.avg_coord_num for cmplx in complexes]
        assert all([7 in acn for acn in complexes_acn])
