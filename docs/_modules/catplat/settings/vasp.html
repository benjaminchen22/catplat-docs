    def cell_optimization_settings(self):
        settings = self.main_settings().delete_incar_parameters(['IOPT'])
        settings = settings.merge_incar_parameters(
            {'LREAL': 'Auto', 'EDIFFG': -0.05, 'PREC': 'N', 'POTIM': 0.05, 'IBRION': 3, 'ISIF': 3})
        settings = settings.delete_incar_parameters(['IOPT'])  # IBRION=3, LREAL=Auto

        kpoints_settings = KpointsSettings.gamma_automatic([3, 3, 3])
        settings = settings.merge_parameters(kpoints_settings=kpoints_settings)
        return settings

    def quick_settings(self):
        # return None
        settings = self.main_settings()
        quick_kpoints_settings = KpointsSettings.gamma_automatic([1, 1, 1])

        settings = settings.merge_incar_parameters({'LREAL': 'Auto'})
        settings = settings.merge_parameters(kpoints_settings=quick_kpoints_settings)
        return [settings]

    def incar_settings(self):
        magmom_dict = {'Ni': 5, 'Fe': 5, 'Co': 5}
        incar = {'LREAL': 'Auto', 'EDIFF': 1E-7, 'IOPT': 1, 'IBRION': 3, 'POTIM': 0, 'LWAVE': 'False', 'PREC': 'A',
                 'EDIFFG': -0.02, 'ICHARG': 1, 'NELM': 150, 'ISYM': 0, 'ALGO': 'Fast', 'ENCUT': 520, 'NSW': 1000,
                 'ISPIN': 2,
                 'GGA': 'PE', 'SIGMA': 0.05, 'LCHARG': 'False', 'ENAUG': 1200, 'NELMIN': 4}

        return IncarSettings(incar=incar, magmom=magmom_dict, no_spin_pol_if_no_mag_elements=True)
