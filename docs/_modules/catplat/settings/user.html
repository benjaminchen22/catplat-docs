    def check_mysql_credentials(self, database=None):
        """ Check if the user has configured their mysql credentials. """
        from pyatoms.database.mysql import MySQLServer

        user = self.get('CATPLAT_MYSQL_USER')
        password = self.get('CATPLAT_MYSQL_PWD')
        host = self.get('CATPLAT_MYSQL_HOST', 'localhost')

        if user is None or password is None:
            raise ValueError('Please configure your mysql credentials by running:\n'
                             'catplat config user --mysql-user <USER> --mysql-pwd <PASSWORD>')

        server = MySQLServer(user=user, password=password, host=host)
        server.connect(database=database)
        logger.info(f'Successfully connected to mysql server: {server}, database: {database}')

    def get_default_run_dir(self):
        run_dir = self.get(self.RUNDIR_KEYWORD)

        if run_dir is None:
            raise ValueError('Could not get database path. Configure catplat by:\n'
                             '1. catplat config user --run-dir <RUN_DIR> # to set a default running directory for all projects\n'
                             '2. catplat config project --calculation-path <CALC_PATH> --database-path <DATABASE_PATH> '
                             '# to set specific calculation and database paths for individual projects')
        return run_dir



    @property
    def projects(self):
        return self.get(self.PROJECT_KEYWORD, {})

    def get_calculation_path(self, project):
        """ Automatically determine folders by searching in the following order:

        1) Determines if project has been specified in user_settings (catplat.yaml)
        2) Uses specified RUN_DIR in catplat.yaml
        """
        projects = self.projects

        if project in projects:
            path = projects[project]['CALCULATION_PATH']
        else:
            path = os.path.join(self.get_default_run_dir(), project, 'calculations')
        return os.path.expanduser(path)

    def get_database_path(self, project):
        projects = self.projects

        if project in projects:
            path = projects[project]['DATABASE_PATH']
        else:
            folder = os.path.join(self.get_default_run_dir(), project, 'database')
            path = os.path.join(folder, 'project.db')
        return os.path.expanduser(path)
