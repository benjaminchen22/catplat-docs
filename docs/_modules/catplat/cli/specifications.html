def check_required_ml_modules(ctx, param, value):
    """ Callback to check if required modules are installed. """
    if value is None or value is False:
        return value

    modules = ['atomistic_ml', 'mace', 'ocpmodels', 'torch_geometric', 'lmdb', 'orjson', 'wandb', 'torch_sparse',
               'torch_scatter']
    missing_modules = []
    for module in modules:
        try:
            __import__(module)
        except ImportError as e:
            if f"No module named: '{module}'" not in str(e):
                raise e
            missing_modules.append(module)

    if missing_modules:
        raise ImportError(f'The following modules are required for the {param.name} option (value={value}) '
                          f'but are not installed: {missing_modules}')

    return value

def validate_nargs(ctx, param, value):
    """
    SpecificationsSet requires (None,) or [None] to rperesent no choice. However, click default will
    only accepts iterable since `nargs` was specified. We manually change the default back to [None]
    in this callback.
    """
    return [None if all(i is None for i in v) else v for v in value]

def bonds_type(s: str):
    """ Custom type for bonds to type cast click string inputs. 

    Valid examples:
    - 0 
    - 0,1 
    - monodentate/bidentate
    """
    s = s.replace(" ", "")
    if s == 'monodentate' or s == 'bidentate':
        return s

    try:
        s = [int(i) for i in s.split(',')]
        return s

    except:
        raise ValueError(f'Unsupported input for --bonds: {s}')

def avg_coord_num_type(s: str):
    """ Custom type for avg_coord_num to type cast click string inputs.

    Valid examples:
    - 7 
    - 7,7 
    - <=8
    - <8,7
    """
    s = s.replace(" ", "")
    try:
        def type_cast(string):
            """ Average coordination can be comparator string, int, or float. """
            from catplat.utils.comparator import Comparator
            from ast import literal_eval

            if Comparator.is_comparator(string):
                return string

            numeric = literal_eval(string)
            if not isinstance(numeric, (int, float)):
                raise ValueError(f'Unsupported input for --avg-coord-num: {s}')
            return numeric

        s = [type_cast(i) for i in s.split(',')]
        return s

    except:
        raise ValueError(f'Unsupported input for --avg-coord-num: {s}')

def click_bulk_options_calculate(f):
    """ Common click options relevant to bulk calculation specifications. """

    @click.option('--bulk-atoms', type=stripped_str, multiple=True, default=[None],
                  help='Name of bulk file in user bulk directory. '
                       'When bulk is specified, other bulk attributes such as e_above_hull should not be specified.')
    @click.option('--chemsys', type=stripped_str, multiple=True, default=[None],
                  help='Chemical species of the bulk. Multiple elements should be separated by dashes. e.g. --chemsys Cu-Rh')
    @click.option('--min-e-above-hull', type=float, default=None,
                  help='Minimum energy above hull of the bulk structure as indicated on Materials Project. [eV/atom]')
    @click.option('--max-e-above-hull', type=float, default=None,
                  help='Maximum energy above hull of the bulk structure as indicated on Materials Project. [eV/atom]')
    @click.option('--bulk-formula', type=stripped_str, multiple=True, default=[None],
                  help='Bulk formula as indicated on Materials Project. e.g. --bulk-formula Cu3Rh')
    @click.option('--spacegroup', type=int, multiple=True, default=[None],
                  help='Spacegroup number of bulk as indicated on Materials Project.')
    @click.option('--mpid', type=stripped_str, multiple=True, default=[None],
                  help='Materials Project ID of bulk structure')
    @functools.wraps(f)
    def wrapper_common_options(*args, **kwargs):
        return f(*args, **kwargs)

    return wrapper_common_options
